5.3 Mantel Test (MANTEL)

(Mantel 1967)

run_MANTEL(enterotypes_arumugam,
           y_variables = c("Enterotype", "Clinical.Status"),
           z_variables = c("Nationality", "Gender"),
           norm = FALSE,
           method = "mantel.partial",
           method_dist = c("bray", "euclidean", "jaccard"))
